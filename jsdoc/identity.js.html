/**
 * Core identity factory.
 *
 * @module core/factories/identity
 * @author rajkissu <rajkissu@gmail.com>
 */
app.factory('OpsIdentity', function (OPS_SERVER, $http, $q, $rootScope) {
  'use strict';

  var _identity, _loggedIn;

  /**
   * Sets up a login identity.
   *
   * @param {Object} identity - the identity object.
   */
  function login(identity) {
    _identity = identity;
    _loggedIn = !!identity;

    if (_loggedIn) {
      // TODO: find a better place to inject this?
      $rootScope.identity = _identity;
    }
  }

  /**
   * Performs a logout.
   *
   * @returns {Object} logout promise.
   */
  function logout() {
    var deferred = $q.defer();

    $http.get(OPS_SERVER + '/logout').success(function (data) {
      login(null);

      deferred.resolve(null);
    });

    return deferred.promise;
  }

  /**
   * Checks if the identity object exists.
   *
   * @returns {Boolean} true if the identity object has been defined.
   */
  function isResolved() {
    return !!_identity;
  }

  /**
   * Checks if the user has logged in.
   *
   * @returns {Boolean} true if the user is currently logged in.
   */
  function isLoggedIn() {
    return _loggedIn;
  }

  /**
   * Validates the current identity.
   *
   * @returns {Object} identity promise.
   */
  function identity() {
    var deferred = $q.defer();

    // if the identity already exists,
    // return the promise immediately
    if (!!_identity) {
      deferred.resolve(_identity);

      return deferred.promise;
    }

    // check the API for a session
    $http.get(OPS_SERVER + '/login')
    .success(function (data) {
      // authorization exists, resolve
      // immediately
      login(data);
      deferred.resolve(_identity);
    })
    .error(function (data) {
      // authorization failed, resolve
      // immediately
      login(null);
      deferred.resolve(_identity);
    });

    return deferred.promise;
  }

  return {
    login      : login,
    logout     : logout,
    isResolved : isResolved,
    isLoggedIn : isLoggedIn,
    identity   : identity
  };
});
