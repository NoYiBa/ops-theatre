/**
 * Core filetree directive.
 *
 * @module core/directives/filetree
 * @author rajkissu <rajkissu@gmail.com>
 */
app.directive('opsFileTree', function ($http, $parse) {
  return {
    // custom element tag
    restrict : 'E',

    // accepts `data` parameter
    scope : {
      data : '='
    },

    link : function (scope, element, attrs) {
      var data, url, onSelect;

      data     = scope.data;
      url      = data.url;
      onSelect = data.onSelect;

      $http.get(url).success(function (files) {
        scope.model = files;

        // monitor tree selection changes
        scope.$watch(function () {
          if (scope.id) {
            return scope.id.currentNode;
          }
        }, function (newObj, oldObj) {
          var selected = scope.id.currentNode;

          // Make sure:
          // 1. there is a selection
          // 2. it's not a directory (0 children)
          // 3. there's a select method to run
          if (selected && !selected.children.length && onSelect) {
            onSelect(selected, data);
          }
        });
      });
    },

    templateUrl : 'modules/core/templates/filetree.html'
  };
});
