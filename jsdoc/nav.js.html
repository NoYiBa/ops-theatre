/**
 * Core navigation factory.
 *
 * @module core/factories/nav
 * @author rajkissu <rajkissu@gmail.com>
 */
app.factory('OpsNav', function ($rootScope) {
  'use strict';

  // the navigations array
  var _opsNav = [];

  /**
   * Adds an entry or multiple entries to the
   * navigations array.
   *
   * @param {Object|Object[]} item - entry to add, single or array of entries.
   */
  function add(item) {
    angular.isArray(item) ?
      angular.forEach(item, _add) :
      _add(item);
  }

  /**
   * Internal method for adding single entries
   * to the navigations array.
   *
   * @private
   *
   * @param {Object} item - the entry to add.
   */
  function _add(item) {
    // check to make sure:
    //
    // 1. the entry has at least a title
    // 2. it doesn't already exist in the array
    if (
      item.title
      && !_opsNav.find(item)
    ) {
      _opsNav.push(item);
    }
  }

  /**
   * Fetches a specific navigation entry.
   *
   * @param {Object} criteria - the criteria to search for the item by.
   * @returns {Object} the item in the navigation array.
   */
  function get(criteria) {
    return _opsNav.find(criteria);
  }

  /**
   * Send the entire navigations array.
   *
   * @returns {Array} the navigation array.
   */
  function all() {
    return _opsNav;
  }

  // add this to the root scope
  $rootScope.opsNav = _opsNav;

  return {
    add : add,
    get : get,
    all : all
  };
});
