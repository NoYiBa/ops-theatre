/**
 * Core authorization factory.
 *
 * @module core/factories/authorize
 * @author rajkissu <rajkissu@gmail.com>
 */
app.factory('OpsAuthorize', function ($rootScope, $state, $location, OpsIdentity) {
  'use strict';

  /**
   * Performs authorization check between state transitions.
   *
   * @returns {Object} authorization promise.
   */
  function authorize(toState) {
    return OpsIdentity.identity()
    .then(function () {
      var isLoggedIn, returnToState, returnToStateParams;

      // if an identity exists, check if it's logged in
      isLoggedIn = OpsIdentity.isLoggedIn();

      if (isLoggedIn) {
        returnToState       = $rootScope.returnToState;
        returnToStateParams = $rootScope.returnToStateParams;

        // proceed as per the norm
        if (returnToState) {
          if (returnToState.name === 'login') {
            // already logged in, go to dashboard
            $state.go('restricted.dashboard');
          } else {
            // go to user-desired state
            $state.go(returnToState, returnToStateParams);

            // clear the state
            $rootScope.returnToState       = null;
            $rootScope.returnToStateParams = null;
          }
        }
      } else {
        $rootScope.returnToState       = $rootScope.toState;
        $rootScope.returnToStateParams = $rootScope.toStateParams;

        // redirect to the login page
        $state.go('login');
      }
    });
  }

  return {
    authorize : authorize
  };
});

/**
 * Intercept requests to check for authorization.
 */
app.run(function ($rootScope, $state, $stateParams, OpsAuthorize, OpsIdentity) {
  $rootScope.$on('$stateChangeStart', function (event, toState, stateParams) {
    $rootScope.toState     = toState;
    $rootScope.stateParams = stateParams;

    // identity has been resolved, do authorization check
    if (OpsIdentity.isResolved()) {
      OpsAuthorize.authorize();
    }
  });
});
