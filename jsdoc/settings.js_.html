/**
 * Core settings controller.
 *
 * @module core/controllers/settings
 * @author rajkissu <rajkissu@gmail.com>
 *
 * @todo cache settings retrieval
 * @todo fix this code
 */
app.controller('SettingsCtrl', function (OPS_SERVER, $http, $rootScope, $scope, $stateParams) {
  'use strict';

  $http.get(OPS_SERVER + '/settings').success(function (settings) {
    var links, module;

    links = Object.keys(settings);

    angular.forEach(links, function (name, index, links) {
      links[index] = {
        name : name,
        href : '/settings/' + name 
      };
    });

    $rootScope.nav = links;

    module     = $stateParams.module;
    thisModule = settings[module];

    if (thisModule) {
      $scope.settings = thisModule;
    }
  });
});
